152template<typename T1>
153arma_warn_unused
154inline
155Col<uword>
156find(const SpBase<typename T1::elem_type,T1>& X, const uword k = 0)
157 {
158 arma_extra_debug_sigprint();
159
160 const SpProxy<T1> P(X.get_ref());
161
162 const uword n_rows = P.get_n_rows();
163 const uword n_nz = P.get_n_nonzero();
164
165 Mat<uword> tmp(n_nz, 1, arma_nozeros_indicator());
166
167 uword* tmp_mem = tmp.memptr();
168
169 typename SpProxy<T1>::const_iterator_type it = P.begin();
170
171 for(uword i=0; i<n_nz; ++i)
172 {
173 const uword index = it.row() + it.col()*n_rows;
174
175 tmp_mem[i] = index;
176
177 ++it;
178 }
179
180 Col<uword> out;
181
182 const uword count = (k == 0) ? uword(n_nz) : uword( (std::min)(n_nz, k) );
183
184 out.steal_mem_col(tmp, count);
185
186 return out;
187 }
