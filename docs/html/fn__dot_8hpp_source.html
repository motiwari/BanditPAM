170 template<typename T1, typename T2>
171 arma_hot
172 inline
173 typename T1::elem_type
174 dot_helper(const SpProxy<T1>& pa, const SpProxy<T2>& pb)
175 {
176 typedef typename T1::elem_type eT;
177
178 // Iterate over both objects and see when they are the same
179 eT result = eT(0);
180
181 typename SpProxy<T1>::const_iterator_type a_it = pa.begin();
182 typename SpProxy<T1>::const_iterator_type a_end = pa.end();
183
184 typename SpProxy<T2>::const_iterator_type b_it = pb.begin();
185 typename SpProxy<T2>::const_iterator_type b_end = pb.end();
186
187 while((a_it != a_end) && (b_it != b_end))
188 {
189 if(a_it == b_it)
190 {
191 result += (*a_it) * (*b_it);
192
193 ++a_it;
194 ++b_it;
195 }
196 else if((a_it.col() < b_it.col()) || ((a_it.col() == b_it.col()) && (a_it.row() < b_it.row())))
197 {
198 // a_it is "behind"
199 ++a_it;
200 }
201 else
202 {
203 // b_it is "behind"
204 ++b_it;
205 }
206 }
207
208 return result;
209 }
210
211 }

220typename
221enable_if2
222 <(is_arma_sparse_type<T1>::value) && (is_arma_sparse_type<T2>::value) && (is_same_type<typename T1::elem_type, typename T2::elem_type>::value),

225dot
226 (
227 const T1& x,
228 const T2& y
229 )
230 {
231 arma_extra_debug_sigprint();
232
233 const SpProxy<T1> pa(x);
234 const SpProxy<T2> pb(y);
235
236 arma_debug_assert_same_size(pa.get_n_rows(), pa.get_n_cols(), pb.get_n_rows(), pb.get_n_cols(), "dot()");
237
238 typedef typename T1::elem_type eT;
239
240 typedef typename SpProxy<T1>::stored_type pa_Q_type;
241 typedef typename SpProxy<T2>::stored_type pb_Q_type;
242
243 if(
244 ( (SpProxy<T1>::use_iterator == false) && (SpProxy<T2>::use_iterator == false) )
245 && ( (is_SpMat<pa_Q_type>::value == true ) && (is_SpMat<pb_Q_type>::value == true ) )
246 )
247 {
248 const unwrap_spmat<pa_Q_type> tmp_a(pa.Q);
249 const unwrap_spmat<pb_Q_type> tmp_b(pb.Q);
250
251 const SpMat<eT>& A = tmp_a.M;
252 const SpMat<eT>& B = tmp_b.M;
253
254 if( &A == &B )
255 {
256 // We can do it directly!
257 return op_dot::direct_dot_arma(A.n_nonzero, A.values, A.values);
258 }
259 else
260 {
261 return priv::dot_helper(pa,pb);
262 }
263 }
264 else
265 {
266 return priv::dot_helper(pa,pb);
267 }

282dot

284 const T1& x,
285 const T2& y
286 )
287 {
288 arma_extra_debug_sigprint();
289
290 const Proxy<T1> pa(x);
291 const SpProxy<T2> pb(y);
292
293 arma_debug_assert_same_size(pa.get_n_rows(), pa.get_n_cols(), pb.get_n_rows(), pb.get_n_cols(), "dot()");
294
295 typedef typename T1::elem_type eT;
296

300 typename SpProxy<T2>::const_iterator_type it_end = pb.end();
301

304 {

307 }
