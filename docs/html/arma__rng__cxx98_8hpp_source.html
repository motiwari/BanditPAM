54arma_inline
55int
56arma_rng_cxx98::randi_val()
57 {
58 #if (RAND_MAX == 32767)
59 {
60 // NOTE: this is a better-than-nothing solution
61 // NOTE: see also arma_rng_cxx98::randi_max_val()
62
63 u32 val1 = u32(std::rand());
64 u32 val2 = u32(std::rand());
65
66 val1 <<= 15;
67
68 return (val1 | val2);
69 }
70 #else
71 {
72 return std::rand();
73 }
74 #endif

88inline
89double
90arma_rng_cxx98::randn_val()
91 {
92 // polar form of the Box-Muller transformation:
93 // http://en.wikipedia.org/wiki/Box-Muller_transformation
94 // http://en.wikipedia.org/wiki/Marsaglia_polar_method
95
96 double tmp1 = double(0);
97 double tmp2 = double(0);
98 double w = double(0);
99
100 do
101 {
102 tmp1 = double(2) * double(randi_val()) * (double(1) / double(randi_max_val())) - double(1);
103 tmp2 = double(2) * double(randi_val()) * (double(1) / double(randi_max_val())) - double(1);
104
105 w = tmp1*tmp1 + tmp2*tmp2;
106 }
107 while( w >= double(1) );
108
109 return double( tmp1 * std::sqrt( (double(-2) * std::log(w)) / w) );
110 }

114template<typename eT>
115inline
116void
117arma_rng_cxx98::randn_dual_val(eT& out1, eT& out2)
118 {
119 // make sure we are internally using at least floats
120 typedef typename promote_type<eT,float>::result eTp;
121
122 eTp tmp1 = eTp(0);
123 eTp tmp2 = eTp(0);
124 eTp w = eTp(0);
125
126 do
127 {
128 tmp1 = eTp(2) * eTp(randi_val()) * (eTp(1) / eTp(randi_max_val())) - eTp(1);
129 tmp2 = eTp(2) * eTp(randi_val()) * (eTp(1) / eTp(randi_max_val())) - eTp(1);
130
131 w = tmp1*tmp1 + tmp2*tmp2;
132 }
133 while( w >= eTp(1) );
134
135 const eTp k = std::sqrt( (eTp(-2) * std::log(w)) / w);
136
137 out1 = eT(tmp1*k);
138 out2 = eT(tmp2*k);
139 }

143template<typename eT>
144inline
145void
146arma_rng_cxx98::randi_fill(eT* mem, const uword N, const int a, const int b)
147 {
148 if( (a == 0) && (b == RAND_MAX) )
149 {
150 for(uword i=0; i<N; ++i)
151 {
152 mem[i] = eT(std::rand());
153 }
154 }
155 else
156 {
157 const uword length = uword(b - a + 1);
158
159 const double scale = double(length) / double(randi_max_val());
160
161 for(uword i=0; i<N; ++i)
162 {
163 mem[i] = eT((std::min)( b, (int( double(randi_val()) * scale ) + a) ));
164 }
165 }
166 }
