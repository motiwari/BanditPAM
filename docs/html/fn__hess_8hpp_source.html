98template<typename T1>
99inline
100bool
101hess
102 (
103 Mat<typename T1::elem_type>& U,
104 Mat<typename T1::elem_type>& H,
105 const Base<typename T1::elem_type,T1>& X,
106 const typename arma_blas_type_only<typename T1::elem_type>::result* junk = nullptr
107 )
108 {
109 arma_extra_debug_sigprint();
110 arma_ignore(junk);
111
112 arma_debug_check( void_ptr(&U) == void_ptr(&H), "hess(): 'U' is an alias of 'H'" );
113
114 typedef typename T1::elem_type eT;
115
116 Col<eT> tao;
117
118 const bool status = auxlib::hess(H, X.get_ref(), tao);
119
120 if(H.n_rows == 0)
121 {
122 U.reset();
123 }
124 else
125 if(H.n_rows == 1)
126 {
127 U.ones(1, 1);
128 }
129 else
130 if(H.n_rows == 2)
131 {
132 U.eye(2, 2);
133 }
134 else
135 {
136 U.eye(size(H));
137
138 Col<eT> v;
139
140 for(uword i=0; i < H.n_rows-2; ++i)
141 {
142 // TODO: generate v in a more efficient manner;
143 // TODO: the .ones() operation is an overkill, as most of v is overwritten afterwards
144
145 v.ones(H.n_rows-i-1);
146
147 v(span(1, H.n_rows-i-2)) = H(span(i+2, H.n_rows-1), i);
148
149 U(span::all, span(i+1, H.n_rows-1)) -= tao(i) * (U(span::all, span(i+1, H.n_rows-1)) * v * v.t());
150 }
151
152 U(span::all, H.n_rows-1) = U(span::all, H.n_rows-1) * (eT(1) - tao(H.n_rows-2));
153
154 for(uword i=0; i < H.n_rows-2; ++i)
155 {
156 H(span(i+2, H.n_rows-1), i).zeros();
157 }
158 }
159
160 if(status == false)
161 {
162 U.soft_reset();
163 H.soft_reset();
164 arma_debug_warn_level(3, "hess(): decomposition failed");
165 }
166
167 return status;
168 }
