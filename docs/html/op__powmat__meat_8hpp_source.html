44inline
45bool
46op_powmat::apply_direct(Mat<typename T1::elem_type>& out, const Base<typename T1::elem_type,T1>& X, const uword y, const bool y_neg)
47 {
48 arma_extra_debug_sigprint();
49
50 typedef typename T1::elem_type eT;
51
52 if(y_neg)
53 {
54 if(y == uword(1))
55 {
56 return op_inv::apply_direct(out, X.get_ref(), "powmat()");
57 }
58 else
59 {
60 Mat<eT> X_inv;
61
62 const bool inv_status = op_inv::apply_direct(X_inv, X.get_ref(), "powmat()");
63
64 if(inv_status == false) { return false; }
65
66 op_powmat::apply_direct_positive(out, X_inv, y);
67 }
68 }
69 else
70 {
71 const quasi_unwrap<T1> U(X.get_ref());
72
73 arma_debug_check( (U.M.is_square() == false), "powmat(): given matrix must be square sized" );
74
75 op_powmat::apply_direct_positive(out, U.M, y);
76 }
77

83template<typename eT>
84inline
85void
86op_powmat::apply_direct_positive(Mat<eT>& out, const Mat<eT>& X, const uword y)
87 {
88 arma_extra_debug_sigprint();
89
90 const uword N = X.n_rows;
91
92 if(y == uword(0)) { out.eye(N,N); return; }
93 if(y == uword(1)) { out = X; return; }
94
95 if(X.is_diagmat())
96 {
97 podarray<eT> tmp(N); // use temporary array in case we have aliasing
98
99 for(uword i=0; i<N; ++i) { tmp[i] = eop_aux::pow(X.at(i,i), int(y)); }
100
101 out.zeros(N,N);
102
103 for(uword i=0; i<N; ++i) { out.at(i,i) = tmp[i]; }
104 }
105 else
106 {
107 if(y == uword(2)) { out = X*X; }
108 else if(y == uword(3)) { const Mat<eT> tmp = X*X; out = X*tmp; }
109 else if(y == uword(4)) { const Mat<eT> tmp = X*X; out = tmp*tmp; }
110 else if(y == uword(5)) { const Mat<eT> tmp = X*X; out = X*tmp*tmp; }
111 else
112 {
113 Mat<eT> tmp = X;
114
115 out = X;
116
117 uword z = y-1;
118
119 while(z > 0)
120 {
121 if(z & 1) { out = tmp * out; }
122
123 z /= uword(2);
124
125 if(z > 0) { tmp = tmp * tmp; }
126 }
127 }
128 }
129 }

155template<typename T1>
156inline
157bool
158op_powmat_cx::apply_direct(Mat< std::complex<typename T1::pod_type> >& out, const Base<typename T1::elem_type,T1>& X, const typename T1::pod_type y)
159 {
160 arma_extra_debug_sigprint();
161
162 typedef typename T1::elem_type in_eT;
163 typedef typename T1::pod_type in_T;
164 typedef std::complex<in_T> out_eT;
165
166 if( y == in_T(int(y)) )
167 {
168 arma_extra_debug_print("op_powmat_cx::apply_direct(): integer exponent detected; redirecting to op_powmat");
169
170 const uword y_val = (y < int(0)) ? uword(-y) : uword(y);
171 const bool y_neg = (y < int(0));
172
173 Mat<in_eT> tmp;
174
175 const bool status = op_powmat::apply_direct(tmp, X.get_ref(), y_val, y_neg);
176
177 if(status == false) { return false; }
178
179 out = conv_to< Mat<out_eT> >::from(tmp);
180
181 return true;
182 }
183
184 const quasi_unwrap<T1> U(X.get_ref());
185 const Mat<in_eT>& A = U.M;
186
187 arma_debug_check( (A.is_square() == false), "powmat(): given matrix must be square sized" );
188
189 const uword N = A.n_rows;
190
191 if(A.is_diagmat())
192 {
193 podarray<out_eT> tmp(N); // use temporary array in case we have aliasing
194
195 for(uword i=0; i<N; ++i) { tmp[i] = eop_aux::pow( std::complex<in_T>(A.at(i,i)), y) ; }
196
197 out.zeros(N,N);
198
199 for(uword i=0; i<N; ++i) { out.at(i,i) = tmp[i]; }
200
201 return true;
202 }
203
204 #if defined(ARMA_OPTIMISE_SYMPD)
205 const bool try_sympd = sympd_helper::guess_sympd_anysize(A);
206 #else
207 const bool try_sympd = false;
208 #endif
209
210 if(try_sympd)
211 {
212 Col<in_T> eigval;
213 Mat<in_eT> eigvec;
214
215 const bool eig_status = eig_sym(eigval, eigvec, A);
216
217 if(eig_status)
218 {
219 eigval = pow(eigval, y);
220
221 const Mat<in_eT> tmp = diagmat(eigval) * eigvec.t();
222
223 out = conv_to< Mat<out_eT> >::from(eigvec * tmp);
224
225 return true;
226 }
227
228 // fallthrough
229 }
230
231 bool powmat_status = false;
232
233 Col<out_eT> eigval;
234 Mat<out_eT> eigvec;
235
236 const bool eig_status = eig_gen(eigval, eigvec, A);
237
238 if(eig_status)
239 {
240 eigval = pow(eigval, y);
241
242 Mat<out_eT> eigvec_t = trans(eigvec);
243 Mat<out_eT> tmp = diagmat(conj(eigval)) * eigvec_t;
244
245 const bool solve_status = auxlib::solve_square_fast(out, eigvec_t, tmp);
246
247 if(solve_status) { out = trans(out); powmat_status = true; }
248 }
249
250 return powmat_status;
251 }
