91template<typename cx_type, bool inverse, uword fixed_N = 0>
92class fft_engine : public fft_store<cx_type, fixed_N, (fixed_N > 0)>
93 {
94 public:
95
96 typedef typename get_pod_type<cx_type>::result T;
97
98 using fft_store<cx_type, fixed_N, (fixed_N > 0)>::N;
99 using fft_store<cx_type, fixed_N, (fixed_N > 0)>::coeffs_ptr;
100
101 podarray<uword> residue;
102 podarray<uword> radix;
103
104 podarray<cx_type> tmp_array;
105
106
107 template<bool fill>
108 inline
109 uword
110 calc_radix()
111 {
112 uword i = 0;
113
114 for(uword n = N, r=4; n >= 2; ++i)
115 {
116 while( (n % r) > 0 )
117 {
118 switch(r)
119 {
120 case 2: r = 3; break;
121 case 4: r = 2; break;
122 default: r += 2; break;
123 }
124
125 if(r*r > n) { r = n; }
126 }
127
128 n /= r;
129
130 if(fill)
131 {
132 residue[i] = n;
133 radix[i] = r;
134 }
135 }
136
137 return i;
138 }

142 inline
143 fft_engine(const uword in_N)
144 : fft_store< cx_type, fixed_N, (fixed_N > 0) >(in_N)
145 {
146 arma_extra_debug_sigprint();
147
148 const uword len = calc_radix<false>();
149
150 residue.set_size(len);
151 radix.set_size(len);
152
153 calc_radix<true>();
154
155
156 // calculate the constant coefficients
157
158 cx_type* coeffs = coeffs_ptr();
159
160 const T k = T( (inverse) ? +2 : -2 ) * std::acos( T(-1) ) / T(N);
161
162 for(uword i=0; i < N; ++i) { coeffs[i] = std::exp( cx_type(T(0), i*k) ); }
163 }

168 inline
169 void
170 butterfly_2(cx_type* Y, const uword stride, const uword m)
171 {
172 arma_extra_debug_sigprint();
173
174 const cx_type* coeffs = coeffs_ptr();
175
176 for(uword i=0; i < m; ++i)
177 {
178 const cx_type t = Y[i+m] * coeffs[i*stride];
179
180 Y[i+m] = Y[i] - t;
181 Y[i ] += t;
182 }
183 }

188 inline
189 void
190 butterfly_3(cx_type* Y, const uword stride, const uword m)
191 {
192 arma_extra_debug_sigprint();
193
194 arma_aligned cx_type tmp[5];
195
196 cx_type* coeffs1 = coeffs_ptr();
197 cx_type* coeffs2 = coeffs1;
198
199 const T coeff_sm_imag = coeffs1[stride*m].imag();
200
201 const uword n = m*2;
202
203 // TODO: rearrange the indices within tmp[] into a more sane order
204
205 for(uword i = m; i > 0; --i)
206 {
207 tmp[1] = Y[m] * (*coeffs1);
208 tmp[2] = Y[n] * (*coeffs2);
209
210 tmp[0] = tmp[1] - tmp[2];
211 tmp[0] *= coeff_sm_imag;
212
213 tmp[3] = tmp[1] + tmp[2];
214
215 Y[m] = cx_type( (Y[0].real() - (T(0.5)*tmp[3].real())), (Y[0].imag() - (T(0.5)*tmp[3].imag())) );
216
217 Y[0] += tmp[3];
218
219
220 Y[n] = cx_type( (Y[m].real() + tmp[0].imag()), (Y[m].imag() - tmp[0].real()) );
221
222 Y[m] += cx_type( -tmp[0].imag(), tmp[0].real() );
223
224 Y++;
225
226 coeffs1 += stride;
227 coeffs2 += stride*2;
228 }
229 }

254
255 //tmp[4] = tmp[0] - tmp[2];

263 tmp[5] = Y[i] - tmp[1];
264
265
266 Y[i ] += tmp[1];
267 Y[i + m2] = Y[i] - tmp[3];
268 Y[i ] += tmp[3];

276 inline
277 arma_hot
278 void
279 butterfly_5(cx_type* Y, const uword stride, const uword m)
280 {
281 arma_extra_debug_sigprint();
282
283 arma_aligned cx_type tmp[13];
284
285 const cx_type* coeffs = coeffs_ptr();
286
287 const T a_real = coeffs[stride*1*m].real();
288 const T a_imag = coeffs[stride*1*m].imag();
289
290 const T b_real = coeffs[stride*2*m].real();
291 const T b_imag = coeffs[stride*2*m].imag();
292
293 cx_type* Y0 = Y;
294 cx_type* Y1 = Y + 1*m;
295 cx_type* Y2 = Y + 2*m;
296 cx_type* Y3 = Y + 3*m;
297 cx_type* Y4 = Y + 4*m;
298
299 for(uword i=0; i < m; ++i)
300 {
301 tmp[0] = (*Y0);
302
303 tmp[1] = (*Y1) * coeffs[stride*1*i];
304 tmp[2] = (*Y2) * coeffs[stride*2*i];
305 tmp[3] = (*Y3) * coeffs[stride*3*i];
306 tmp[4] = (*Y4) * coeffs[stride*4*i];
307
308 tmp[7] = tmp[1] + tmp[4];
309 tmp[8] = tmp[2] + tmp[3];
310 tmp[9] = tmp[2] - tmp[3];
311 tmp[10] = tmp[1] - tmp[4];
312
313 (*Y0) += tmp[7];
314 (*Y0) += tmp[8];
315
316 tmp[5] = tmp[0] + cx_type( ( (tmp[7].real() * a_real) + (tmp[8].real() * b_real) ), ( (tmp[7].imag() * a_real) + (tmp[8].imag() * b_real) ) );
317
318 tmp[6] = cx_type( ( (tmp[10].imag() * a_imag) + (tmp[9].imag() * b_imag) ), ( -(tmp[10].real() * a_imag) - (tmp[9].real() * b_imag) ) );
319
320 (*Y1) = tmp[5] - tmp[6];
321 (*Y4) = tmp[5] + tmp[6];
322
323 tmp[11] = tmp[0] + cx_type( ( (tmp[7].real() * b_real) + (tmp[8].real() * a_real) ), ( (tmp[7].imag() * b_real) + (tmp[8].imag() * a_real) ) );
324
325 tmp[12] = cx_type( ( -(tmp[10].imag() * b_imag) + (tmp[9].imag() * a_imag) ), ( (tmp[10].real() * b_imag) - (tmp[9].real() * a_imag) ) );
326
327 (*Y2) = tmp[11] + tmp[12];
328 (*Y3) = tmp[11] - tmp[12];
329
330 Y0++;
331 Y1++;
332 Y2++;
333 Y3++;
334 Y4++;
335 }
336 }

340 arma_hot
341 inline
342 void
343 butterfly_N(cx_type* Y, const uword stride, const uword m, const uword r)
344 {
345 arma_extra_debug_sigprint();
346
347 const cx_type* coeffs = coeffs_ptr();
348
349 tmp_array.set_min_size(r);
350 cx_type* tmp = tmp_array.memptr();
351
352 for(uword u=0; u < m; ++u)
353 {
354 uword k = u;
355
356 for(uword v=0; v < r; ++v)
357 {
358 tmp[v] = Y[k];
359 k += m;
360 }
361
362 k = u;
363
364 for(uword v=0; v < r; ++v)
365 {
366 Y[k] = tmp[0];
367
368 uword j = 0;
369
370 for(uword w=1; w < r; ++w)
371 {
372 j += stride * k;
373
374 if(j >= N) { j -= N; }
375
376 Y[k] += tmp[w] * coeffs[j];
377 }
378
379 k += m;
380 }
381 }
382 }

386 inline
387 void
388 run(cx_type* Y, const cx_type* X, const uword stage = 0, const uword stride = 1)
389 {
390 arma_extra_debug_sigprint();
391
392 const uword m = residue[stage];
393 const uword r = radix[stage];
394
395 const cx_type *Y_end = Y + r*m;
396
397 if(m == 1)
398 {
399 for(cx_type* Yi = Y; Yi != Y_end; Yi++, X += stride) { (*Yi) = (*X); }
400 }
401 else
402 {
403 const uword next_stage = stage + 1;
404 const uword next_stride = stride * r;
405
406 for(cx_type* Yi = Y; Yi != Y_end; Yi += m, X += stride) { run(Yi, X, next_stage, next_stride); }
407 }
408
409 switch(r)
410 {
411 case 2: butterfly_2(Y, stride, m ); break;
412 case 3: butterfly_3(Y, stride, m ); break;
413 case 4: butterfly_4(Y, stride, m ); break;
414 case 5: butterfly_5(Y, stride, m ); break;
415 default: butterfly_N(Y, stride, m, r); break;
416 }
417 }
