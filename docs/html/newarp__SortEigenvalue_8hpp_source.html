21// When comparing eigenvalues, we first calculate the "target" to sort.
22// For example, if we want to choose the eigenvalues with largest magnitude, the target will be -std::abs(x).
23// The minus sign is due to the fact that std::sort() sorts in ascending order.
24
25
26// default target: throw an exception
27template<typename eT, int SelectionRule>
28struct SortingTarget
29 {
30 arma_inline static typename get_pod_type<eT>::result get(const eT& val)
31 {
32 arma_ignore(val);
33 arma_stop_logic_error("newarp::SortingTarget: incompatible selection rule");
34
35 typedef typename get_pod_type<eT>::result out_T;
36 return out_T(0);
37 }
38 };

142// Sort eigenvalues and return the order index
143template<typename PairType>
144struct PairComparator
145 {
146 arma_inline bool operator() (const PairType& v1, const PairType& v2)
147 {
148 return v1.first < v2.first;
149 }
150 };

153template<typename eT, int SelectionRule>
154class SortEigenvalue
155 {
156 private:
157
158 typedef typename get_pod_type<eT>::result TargetType; // type of the sorting target, will be a floating number type, eg. double
159 typedef std::pair<TargetType, uword> PairType; // type of the sorting pair, including the sorting target and the index
160
161 std::vector<PairType> pair_sort;
162
163
164 public:
165
166 inline
167 SortEigenvalue(const eT* start, const uword size)
168 : pair_sort(size)
169 {
170 arma_extra_debug_sigprint();
171
172 for(uword i = 0; i < size; i++)
173 {
174 pair_sort[i].first = SortingTarget<eT, SelectionRule>::get(start[i]);
175 pair_sort[i].second = i;
176 }
177
178 PairComparator<PairType> comp;
179
180 std::sort(pair_sort.begin(), pair_sort.end(), comp);
181 }
182
183
184 inline
185 std::vector<uword>
186 index()
187 {
188 arma_extra_debug_sigprint();
189
190 const uword len = pair_sort.size();
191
192 std::vector<uword> ind(len);
193
194 for(uword i = 0; i < len; i++) { ind[i] = pair_sort[i].second; }
195
196 return ind;
197 }
198 };
