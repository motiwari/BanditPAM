21template<typename eT>
22inline
23void
24DoubleShiftQR<eT>::compute_reflector(const eT& x1, const eT& x2, const eT& x3, uword ind)
25 {
26 arma_extra_debug_sigprint();
27
28 // In general case the reflector affects 3 rows
29 ref_nr(ind) = 3;
30 eT x2x3 = eT(0);
31 // If x3 is zero, decrease nr by 1
32 if(std::abs(x3) < prec)
33 {
34 // If x2 is also zero, nr will be 1, and we can exit this function
35 if(std::abs(x2) < prec)
36 {
37 ref_nr(ind) = 1;
38 return;
39 }
40 else
41 {
42 ref_nr(ind) = 2;
43 }
44 x2x3 = std::abs(x2);
45 }
46 else
47 {
48 x2x3 = arma_hypot(x2, x3);
49 }
50
51 // x1' = x1 - rho * ||x||
52 // rho = -sign(x1), if x1 == 0, we choose rho = 1
53 eT x1_new = x1 - ((x1 <= 0) - (x1 > 0)) * arma_hypot(x1, x2x3);
54 eT x_norm = arma_hypot(x1_new, x2x3);
55 // Double check the norm of new x
56 if(x_norm < prec)
57 {
58 ref_nr(ind) = 1;
59 return;
60 }
61 ref_u(0, ind) = x1_new / x_norm;
62 ref_u(1, ind) = x2 / x_norm;
63 ref_u(2, ind) = x3 / x_norm;
64 }

79template<typename eT>
80inline
81void
82DoubleShiftQR<eT>::update_block(uword il, uword iu)
83 {
84 arma_extra_debug_sigprint();
85
86 // Block size
87 uword bsize = iu - il + 1;
88
89 // If block size == 1, there is no need to apply reflectors
90 if(bsize == 1)
91 {
92 ref_nr(il) = 1;
93 return;
94 }
95
96 // For block size == 2, do a Givens rotation on M = X * X - s * X + t * I
97 if(bsize == 2)
98 {
99 // m00 = x00 * (x00 - s) + x01 * x10 + t
100 eT m00 = mat_H(il, il) * (mat_H(il, il) - shift_s) +
101 mat_H(il, il + 1) * mat_H(il + 1, il) +
102 shift_t;
103 // m10 = x10 * (x00 + x11 - s)
104 eT m10 = mat_H(il + 1, il) * (mat_H(il, il) + mat_H(il + 1, il + 1) - shift_s);
105 // This causes nr=2
106 compute_reflector(m00, m10, 0, il);
107 // Apply the reflector to X
108 apply_PX(mat_H, il, il, 2, n - il, il);
109 apply_XP(mat_H, 0, il, il + 2, 2, il);
110
111 ref_nr(il + 1) = 1;
112 return;
113 }
114
115 // For block size >=3, use the regular strategy
116 eT m00 = mat_H(il, il) * (mat_H(il, il) - shift_s) +
117 mat_H(il, il + 1) * mat_H(il + 1, il) +
118 shift_t;
119 eT m10 = mat_H(il + 1, il) * (mat_H(il, il) + mat_H(il + 1, il + 1) - shift_s);
120 // m20 = x21 * x10
121 eT m20 = mat_H(il + 2, il + 1) * mat_H(il + 1, il);
122 compute_reflector(m00, m10, m20, il);
123
124 // Apply the first reflector
125 apply_PX(mat_H, il, il, 3, n - il, il);
126 apply_XP(mat_H, 0, il, il + (std::min)(bsize, uword(4)), 3, il);
127
128 // Calculate the following reflectors
129 // If entering this loop, block size is at least 4.
130 for(uword i = 1; i < bsize - 2; i++)
131 {
132 compute_reflector(mat_H.colptr(il + i - 1) + il + i, il + i);
133 // Apply the reflector to X
134 apply_PX(mat_H, il + i, il + i - 1, 3, n + 1 - il - i, il + i);
135 apply_XP(mat_H, 0, il + i, il + (std::min)(bsize, uword(i + 4)), 3, il + i);
136 }
137
138 // The last reflector
139 // This causes nr=2
140 compute_reflector(mat_H(iu - 1, iu - 2), mat_H(iu, iu - 2), 0, iu - 1);
141 // Apply the reflector to X
142 apply_PX(mat_H, iu - 1, iu - 2, 2, n + 2 - iu, iu - 1);
143 apply_XP(mat_H, 0, iu - 1, il + bsize, 2, iu - 1);
144
145 ref_nr(iu) = 1;
146 }

150template<typename eT>
151inline
152void
153DoubleShiftQR<eT>::apply_PX(Mat<eT>& X, uword oi, uword oj, uword nrow, uword ncol, uword u_ind)
154 {
155 arma_extra_debug_sigprint();
156
157 if(ref_nr(u_ind) == 1) { return; }
158
159 // Householder reflectors at index u_ind
160 Col<eT> u(ref_u.colptr(u_ind), 3, false);
161
162 const uword stride = X.n_rows;
163 const eT u0_2 = 2 * u(0);
164 const eT u1_2 = 2 * u(1);
165
166 eT* xptr = &X(oi, oj);
167 if(ref_nr(u_ind) == 2 || nrow == 2)
168 {
169 for(uword i = 0; i < ncol; i++, xptr += stride)
170 {
171 eT tmp = u0_2 * xptr[0] + u1_2 * xptr[1];
172 xptr[0] -= tmp * u(0);
173 xptr[1] -= tmp * u(1);
174 }
175 }
176 else
177 {
178 const eT u2_2 = 2 * u(2);
179 for(uword i = 0; i < ncol; i++, xptr += stride)
180 {
181 eT tmp = u0_2 * xptr[0] + u1_2 * xptr[1] + u2_2 * xptr[2];
182 xptr[0] -= tmp * u(0);
183 xptr[1] -= tmp * u(1);
184 xptr[2] -= tmp * u(2);
185 }
186 }
187 }

191template<typename eT>
192inline
193void
194DoubleShiftQR<eT>::apply_PX(eT* x, uword u_ind)
195 {
196 arma_extra_debug_sigprint();
197
198 if(ref_nr(u_ind) == 1) { return; }
199
200 eT u0 = ref_u(0, u_ind),
201 u1 = ref_u(1, u_ind),
202 u2 = ref_u(2, u_ind);
203
204 // When the reflector only contains two elements, u2 has been set to zero
205 bool nr_is_2 = (ref_nr(u_ind) == 2);
206 eT dot2 = x[0] * u0 + x[1] * u1 + (nr_is_2 ? 0 : (x[2] * u2));
207 dot2 *= 2;
208 x[0] -= dot2 * u0;
209 x[1] -= dot2 * u1;
210 if(!nr_is_2) { x[2] -= dot2 * u2; }
211 }

215template<typename eT>
216inline
217void
218DoubleShiftQR<eT>::apply_XP(Mat<eT>& X, uword oi, uword oj, uword nrow, uword ncol, uword u_ind)
219 {
220 arma_extra_debug_sigprint();
221
222 if(ref_nr(u_ind) == 1) { return; }
223
224 // Householder reflectors at index u_ind
225 Col<eT> u(ref_u.colptr(u_ind), 3, false);
226 uword stride = X.n_rows;
227 const eT u0_2 = 2 * u(0);
228 const eT u1_2 = 2 * u(1);
229 eT* X0 = &X(oi, oj);
230 eT* X1 = X0 + stride; // X0 => X(oi, oj), X1 => X(oi, oj + 1)
231
232 if(ref_nr(u_ind) == 2 || ncol == 2)
233 {
234 // tmp = 2 * u0 * X0 + 2 * u1 * X1
235 // X0 => X0 - u0 * tmp
236 // X1 => X1 - u1 * tmp
237 for(uword i = 0; i < nrow; i++)
238 {
239 eT tmp = u0_2 * X0[i] + u1_2 * X1[i];
240 X0[i] -= tmp * u(0);
241 X1[i] -= tmp * u(1);
242 }
243 }
244 else
245 {
246 eT* X2 = X1 + stride; // X2 => X(oi, oj + 2)
247 const eT u2_2 = 2 * u(2);
248 for(uword i = 0; i < nrow; i++)
249 {
250 eT tmp = u0_2 * X0[i] + u1_2 * X1[i] + u2_2 * X2[i];
251 X0[i] -= tmp * u(0);
252 X1[i] -= tmp * u(1);
253 X2[i] -= tmp * u(2);
254 }
255 }
256 }

295template<typename eT>
296void
297DoubleShiftQR<eT>::compute(const Mat<eT>& mat_obj, eT s, eT t)
298 {
299 arma_extra_debug_sigprint();
300
301 arma_debug_check( (mat_obj.is_square() == false), "newarp::DoubleShiftQR::compute(): matrix must be square" );
302
303 n = mat_obj.n_rows;
304 mat_H.set_size(n, n);
305 shift_s = s;
306 shift_t = t;
307 ref_u.set_size(3, n);
308 ref_nr.set_size(n);
309
310 // Make a copy of mat_obj
311 mat_H = mat_obj;
312
313 // Obtain the indices of zero elements in the subdiagonal,
314 // so that H can be divided into several blocks
315 std::vector<uword> zero_ind;
316 zero_ind.reserve(n - 1);
317 zero_ind.push_back(0);
318 eT* Hii = mat_H.memptr();
319 for(uword i = 0; i < n - 2; i++, Hii += (n + 1))
320 {
321 // Hii[1] => mat_H(i + 1, i)
322 const eT h = std::abs(Hii[1]);
323 if(h <= eps_abs || h <= eps_rel * (std::abs(Hii[0]) + std::abs(Hii[n + 1])))
324 {
325 Hii[1] = 0;
326 zero_ind.push_back(i + 1);
327 }
328 // Make sure mat_H is upper Hessenberg
329 // Zero the elements below mat_H(i + 1, i)
330 std::fill(Hii + 2, Hii + n - i, eT(0));
331 }
332 zero_ind.push_back(n);
333
334 for(std::vector<uword>::size_type i = 0; i < zero_ind.size() - 1; i++)
335 {
336 uword start = zero_ind[i];
337 uword end = zero_ind[i + 1] - 1;
338 // Compute refelctors from each block X
339 update_block(start, end);
340 }
341
342 computed = true;
343 }
