23template<typename T1>
24inline
25bool
26op_unique::apply_helper(Mat<typename T1::elem_type>& out, const Proxy<T1>& P, const bool P_is_row)
27 {
28 arma_extra_debug_sigprint();
29
30 typedef typename T1::elem_type eT;
31
32 const uword n_elem = P.get_n_elem();
33
34 if(n_elem == 0)
35 {
36 if(P_is_row)
37 {
38 out.set_size(1,0);
39 }
40 else
41 {
42 out.set_size(0,1);
43 }
44
45 return true;
46 }
47
48 if(n_elem == 1)
49 {
50 const eT tmp = (Proxy<T1>::use_at) ? P.at(0,0) : P[0];
51
52 out.set_size(1, 1);
53
54 out[0] = tmp;
55
56 return true;
57 }
58
59 Mat<eT> X(n_elem, 1, arma_nozeros_indicator());
60
61 eT* X_mem = X.memptr();
62
63 if(Proxy<T1>::use_at == false)
64 {
65 typename Proxy<T1>::ea_type Pea = P.get_ea();
66
67 for(uword i=0; i<n_elem; ++i)
68 {
69 const eT val = Pea[i];
70
71 if(arma_isnan(val)) { out.soft_reset(); return false; }
72
73 X_mem[i] = val;
74 }
75 }
76 else
77 {
78 const uword n_rows = P.get_n_rows();
79 const uword n_cols = P.get_n_cols();
80
81 for(uword col=0; col < n_cols; ++col)
82 for(uword row=0; row < n_rows; ++row)
83 {
84 const eT val = P.at(row,col);
85
86 if(arma_isnan(val)) { out.soft_reset(); return false; }
87
88 (*X_mem) = val; X_mem++;
89 }
90
91 X_mem = X.memptr();
92 }
93
94 arma_unique_comparator<eT> comparator;
95
96 std::sort( X.begin(), X.end(), comparator );
97
98 uword N_unique = 1;
99
100 for(uword i=1; i < n_elem; ++i)
101 {
102 const eT a = X_mem[i-1];
103 const eT b = X_mem[i ];
104
105 const eT diff = a - b;
106
107 if(diff != eT(0)) { ++N_unique; }
108 }
109
110 if(P_is_row)
111 {
112 out.set_size(1, N_unique);
113 }
114 else
115 {
116 out.set_size(N_unique, 1);
117 }
118
119 eT* out_mem = out.memptr();
120
121 if(n_elem > 0) { (*out_mem) = X_mem[0]; out_mem++; }
122
123 for(uword i=1; i < n_elem; ++i)
124 {
125 const eT a = X_mem[i-1];
126 const eT b = X_mem[i ];
127
128 const eT diff = a - b;
129
130 if(diff != eT(0)) { (*out_mem) = b; out_mem++; }
131 }
132
133 return true;
134 }
